```cpp
#include <bits/stdc++.h>
#define ll long long 
using namespace std;

ll dp[5010][5010]={0};

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    
    ll t;
    cin >> t;
    
    while(t--) {
        string A;
        cin >> A;
        ll n = A.length(), ans=0;
        
        dp[0][0]=0;
        for (ll i=1; i<n; i++) {
            if (A[i]==A[0]) {
                dp[i][0]=1;
                dp[0][i]=1;
            } else {
                dp[i][0]=0;
                dp[0][i]=0;
            }
        }
        
        for (ll i=1; i<n; i++) {
            for (ll j=1; j<n; j++) {
                if (A[i]==A[j] && i!=j) dp[i][j]=1+dp[i-1][j-1];
                else dp[i][j]=0;
            }
        }
        
        for (ll i=0; i<n; i++) {
            for (ll j=0; j<n; j++) {
                ans=max(ans, dp[i][j]);
            }
        }
        
        cout << ans << '\n';
    }
    
    
    return 0;
}
```
